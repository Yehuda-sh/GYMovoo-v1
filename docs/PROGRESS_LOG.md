```typescript
// src/services/questionnaireService.ts - לוגיקה חדשה
export function extractEquipmentFromQuestionnaire(
  questionnaire: any
): string[] {
  const equipment: string[] = [];

  // חילוץ ציוד מהשאלות הדינמיות
  const dynamicQuestions = [
    "bodyweight_equipment_options",
    "home_equipment_options",
    "gym_equipment_options",
  ];

  dynamicQuestions.forEach((questionKey) => {
    if (questionnaire[questionKey]?.length > 0) {
      questionnaire[questionKey].forEach((item) => {
        if (item.metadata?.equipmentIds?.length > 0) {
          equipment.push(...item.metadata.equipmentIds);
        }
      });
    }
  });

  return [...new Set(equipment)];
}
```
